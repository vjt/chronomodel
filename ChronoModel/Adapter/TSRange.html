def initialize_type_map(m = type_map)
  super.tap do
    typ = ChronoModel::Adapter::TSRange::Type
    oid = typ::OID

    ar_type = type_map.fetch(oid)
    cm_type = typ.new(ar_type.subtype, ar_type.type)

    type_map.register_type oid, cm_type
  end
end
